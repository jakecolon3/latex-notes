\chapter{Equazioni differenziali}
\section{Definizioni e basi}
\subsubsection{Lezione 9}

\begin{definizione}
Le equazioni differenziali sono equazioni in cui l'incognita è \textit{una funzione}:
\begin{equation*}
    y(t): I \subseteq \mathbb{R} \to \mathbb{R}
\end{equation*}
\end{definizione}

\leavevmode \\
Questa funzione (?) risolve un'equazione in cui entrano in gioco le \textit{derivate} di $y(t)$. Il grado dell'equazione è la derivata di ordine più alto in cui compare $y(t)$.

\begin{esempio}[Equazione di grado I]
    \begin{equation*}
        y'(t) = \cos(y(t)) + y(t) e^t
    \end{equation*}
\end{esempio}
\begin{esempio}[Equazione di grado II]
    \begin{equation*}
        y'' = y
    \end{equation*}
    Ha tra le sue soluzioni $e^x$ e $e^{-x}$.
\end{esempio}

\begin{definizione}
    Un equazione è \textit{normale} se ha questa forma:
    \begin{equation*}
        y^{(n)} = f(t, y(t), y'(t), y''(t), ..., y^{n-1}(t)).
    \end{equation*}
    Le equazioni normali hanno sicuramente soluzione.
\end{definizione}

\subsection{Risolvere eq. Differenziali}
Prendiamo l'equazione:
\begin{equation*}
    y'(t)=f(t)
\end{equation*}
Questa è un equazione differenziale in forma normale, e sappiamo già risolverla: basta trovare una primitiva.
\begin{gather*}
    y'(t) = \cos(t) \\
    y(t) = \int \cos(t)dt \\
    = \sin(t) + c
\end{gather*}
Ma questa non è l'unica soluzione; infatti quest'equazione ha \textit{infinite} soluzioni.

\begin{esempio}
    \begin{gather*}
        y'(t) = e^{-t^2} \\
        y(t) = \int_{a}^{t} e^{-s^2} ds
    \end{gather*}
    Questa equazione ha infinite soluzioni al variare di $a$.
\end{esempio}

\begin{osservazione}
    In generale, le soluzioni delle equazioni differenziali non sono uniche.
\end{osservazione}

\begin{proprieta}
Per avere unicità per equazioni differenziali del primo ordine in forma normale, serve una condizione iniziale:
\begin{equation}
    \begin{cases}
        y't = f(t, y(t)) \\
        y(t_0) = y_0
    \end{cases}
\end{equation}
In questo caso, se $f$ è regolare esiste una soluzione ed è unica.
\end{proprieta}

\begin{esempio}
    \begin{equation}
        \begin{cases}
            y' = \cos(t) \\
            y(0) = 1
        \end{cases}
    \end{equation}
    Ha 1 soluzione: $\sin(t)+1$.
\end{esempio}

\begin{proprieta}
    $y(t)$ è una quantità che aumenta e diminuisce in proporzione a se stessa. Cioè la sua variazione nel tempo è proporzionale al suo valore precedente (?):
    \begin{gather*}
        y(t+\Delta t) - y(t) = \lambda y(t)\Delta t \\
        \frac{y(t+\Delta t) - y(t)}{\Delta t} = \lambda y(t) \\
        \xrightarrow{\Delta t \to 0} y'(t) = \lambda y(t)
    \end{gather*}
    Questa proprietà ha molte applicazioni (es. crescita di una popolazione).
\end{proprieta}

\begin{esempio}
    \begin{equation*}
        y(t) = e^{\lambda t} \\
    \end{equation*}
    La derivata di questa funzione risolve l'equazione differenziale:
    \begin{equation*}
        y'(t) = \lambda e^{\lambda t} = \lambda y(t)
    \end{equation*}
    Ma anche questa ha infinite soluzioni. Vediamo quindi che per far senso delle eq. differenziali ci serve una condizione di partenza:
    \begin{gather*}
        \begin{cases}
            y'(t) = 5y(t) \\
            y(1) = 2
        \end{cases} \\
        y(t) = k^{5t} \\
        y(1) = 2 \\
        ke^2 = 2 \\
        k =  \frac{2}{e^5} \\
        k = 2e^{-5} \\
        y(t) = 2e^{-5}e^{5t}.
    \end{gather*}
\end{esempio}

Vediamo il caso particolare in cui $y(0) = y_0$:
\begin{gather*}
    \begin{cases}
    y' = \lambda y \\
    y(0) = y_0
    \end{cases} \\
    y(t) = ke^{\lambda t} \\
    y(0) = 0 \\
    y(t) = ke^0 = k \\
    k = y_0 \\
    y(t) = y_0 e^{\lambda t} ??
\end{gather*}

Vediamo ora il caso in cui la costante di crescita è una funzione:
\begin{gather*}
    y'(t) = \lambda(t) y(t) \\
    y(t + \Delta t) - y(t) = \lambda(t)y(t)\Delta t
\end{gather*}
Ma come la risolviamo? Dobbiamo trovare una funzione la cui derivata contiene se stessa e la variabile:
\begin{equation*}
    y'(t) = ty(t)
\end{equation*}
Sicuramente sarà del tipo
\begin{equation*}
    y(t) = e^{\Lambda (t)}
\end{equation*}
Ma $\Lambda$ deve essere una primitiva di $t$:
\begin{gather*}
    \Lambda(t) = \int \lambda(t) dt \\
    = \frac{t^2}{2} \to \\
    \to y(t) = e^{\frac{t^2}{2}}
\end{gather*}
Quindi:
\begin{gather}
    y(t) = k e^{\Lambda(t)} \\
    \Lambda(t) = \int \lambda(t) dt.
\end{gather}
Per risolvere $y'(t) = \lambda(t)y(t)$:
\begin{align}
    &\begin{cases}
        y'(t) = \lambda(t)y(t) \\
        y(t_0) = y_0
    \end{cases} \\
    &\begin{cases}
        y(t) = ke^{\Lambda(t)} \\
        y(t_0) = y_0 \to \text{ trova } k
    \end{cases} \\
    &\text{Con } \Lambda(t) = \int \lambda(t)dt
\end{align}

\begin{dimostrazione}
    Dimostriamo che questa è l'unica soluzione.
    \begin{equation*}
        y(t) = ke^{\Lambda(t)}
    \end{equation*}
    Sia $w(t)$ una qualsiasi soluzione. Abbiamo:
    \begin{gather*}
        f(t) = w(t)e^{-\Lambda(t)} \\
        f'(t) = w'(t)e^{-\Lambda(t)} + w(t)(-\Lambda'(t))e^{-\Lambda(t)} = \\
        = e^{-\Lambda(t)}[w'(t)-\lambda(t)w(t)] = 0 \text{ mi sono perso perché} \\
        f'(t) = 0 \to f(t) = k \\
        k = w(t)e^{-\Lambda(t)} \\
        w(t) = ke^{\Lambda(t)}.
    \end{gather*}
\end{dimostrazione}

\begin{definizione}
    Le equazioni della forma
    \begin{equation*}
        y'(t) = \lambda(t)y(t)
    \end{equation*}
    Sono dette \textit{equazioni lineari omogenee}, e si risolvono come abbiamo visto:
    \begin{gather*}
        y(t) = ke^{\Lambda(t)} \\
        \Lambda(t) = \int \lambda(t)dt
    \end{gather*}
\end{definizione}

\begin{definizione}
    Un equazione è \textit{lineare non omogenea} se è del tipo:
    \begin{equation}
        y' = \lambda(t)y(t) + b(t)
    \end{equation}
    Con $b(t) \not= 0$.
\end{definizione}
Sia $\bar{y}(t)$ una soluzione di (4.). Presa una qualsiasi altra soluzione $y(t)$, cosa soddisfa $y(t) = \bar{y}(t)$?
\begin{gather*}
    w(t) = y(t) - \bar{y}(t) \\
    ... \\
    y'(t) = \lambda(t)(y(t)-\bar{y}(t)) = \lambda(t) w(t)
\end{gather*}
Che sappiamo risolvere: $w(t) = ke^{\Lambda(t)} = y(t) - \bar{y}(t)$. Possiamo quindi trovare un'altra soluzione di un eq. omogenea se ne abbiamo una prima.

\subsubsection{Lezione 10}
\subsection{Risolvere equazioni non omogenee}
\begin{definizione}
    Un'equazione differenziale del primo ordine è lineare *non omogenea* se è del tipo:
    \begin{equation}
        y' = \lambda(t) y + b(t)
    \end{equation}
    Con $b(t) \not= 0$.
\end{definizione}

Tutte le soluzioni si ottengono con:
\begin{equation}
    y(t) = ke^{\Lambda(t)} + \bar{y}(t)
\end{equation}
Dove $\bar{y}(t)$ è una soluzione particolare dell'eq. Per ottenere una soluzione particolare si ???

\begin{problema}
    Determinare $B(t)$ t.c.
    \begin{align*}
        &\bar{y}(t) = B(t) e^{\Lambda(t)} \\
        &\bar{y'}(t) = B'(t) e^{\Lambda(t)} + B(t) \Lambda'(t) e^{\Lambda(t)} \\
        =& B'(t) e^{\Lambda(t)} + B(t) \lambda(t)e^{\Lambda(t)} \\
        =& \lambda(t)(B(t)e^{\Lambda(t)}) + b(t) \\
        & B'(t)e^{\Lambda(t)} = b(t) \\
        & B'(t) = \frac{b(t)}{e^{\Lambda(t)}} = b(t)e^{-\Lambda(t)} \\
        & B(t) = \int b(t)e^{-\Lambda(t)} \\
    \end{align*}
\end{problema}

Soluzione generale:
\begin{equation}
    y(t) = ke^{\Lambda(t)} + B(t)e^{\Lambda(t)} = e^{\Lambda(t)}[k+B(t)]\ \ \forall k\in \mathbb{R}
\end{equation}

Quindi per risolvere $y' = \lambda(t) \cdot y + b(t)$ servono solo 2 integrali:
\begin{enumerate}
    \item Si trova $\Lambda(t) = \int \lambda(t) dt$
    \item Si trova $B(t) = \int b(t)e^{-\Lambda(t)} dt$
    \item Le soluzioni sono $y(t) = e^{\Lambda(t)}[k+B(t)]\ \ \forall k\in \mathbb{R}$
\end{enumerate}

\subsubsection{Metodo di somiglianza}
C'è un altro metodo per trovare $\bar{y}(t)$, applicabile se $\lambda(t) = \lambda$, cioè se $\lambda$ è *costante*, e se $b(t)$ è un polinomio, un esponenziale o un seno/coseno. Questo metodo non richede di calcolare integrali:
\begin{enumerate}
    \item Se $\lambda \not = 0$ e $b(t)$ è un polinomio di grado $m$ allora cerco $\bar{y}(t)$ come polinomio dello stesso grado. (non so come spiegarlo ha fatto un esempio) \\
        (sostituisci il $\bar{y}$ trovato nell'eq iniziale)
    \item Se $b(t) = Me^{mt}$ e $\lambda \not= m$, con $\lambda, m, M$ fissati allora $\bar{y}(t)=c_1e^{mt}$.
    \item Se $y' = \lambda y + M \cos(\alpha t)$ e $b(t) = \cos(\alpha t)$ o $\sin(\alpha t)$ allora $\bar{y}(t) = c_1\cos(\alpha t) + c_2\sin(\alpha t)$.
\end{enumerate}

\section{Equazioni Lineari del Secondo Grado}
\begin{definizione}
    Le equazioni lineari del secondo grado sono del tipo:
    \begin{equation}
        y'' + a(t)y'(t)+b(t)y(t) = f(t).
    \end{equation}
    Se $f(t)\equiv 0$ si dice lineare omogenea.
\end{definizione}
\leavevmode \\
Cercheremo le soluzioni di eq. II grado dove $a(t)$ e $b(t)$ sono costanti.
\begin{proprieta}
    Se $a,b,f \in C(I)$ e $f(t)\equiv 0$, in questo caso se si trovano $y_1$ e $y_2$ che soddisfano l'eq. omogenea e non sono proporzionali (la "stessa" soluzione), tutte le soluzioni dell'eq. omogenea sono del tipo
    \begin{equation}
        y(t) = c_1y_1(t) + c_2y_2(t)\ \ \text{ con $c_1, c_2$ costanti generiche.}
    \end{equation}
\end{proprieta}

\subsection{Risolvere eq. II grado}
\subsubsection{Equazioni non omogenee}
Se $f(t) \not \equiv 0$ allore bisogna cercare $\bar{y}(t)$ soluzione particolare dell'eq. completa. Trovata $\bar{y}$, tutte le soluzioni saranno del tipo
\begin{equation}
    y''(t)=c_1y_1(t)+c_2y_2(t) + \bar{y}(t)\ \ c_1, c_2\in \mathbb{R}
\end{equation}
dove $y_1, y_2$ soddisfano l'eq. omogenea associata, ovvero:
\begin{equation}
    y''(t) + c(t)y'(t) + b(t)y(t) = 0
\end{equation}
e $\bar{y}$ è una soluzione particolare dell'equazione.
\begin{osservazione}
    Nelle eq. di I grado per trovare una soluzione particolare bastava imporre 1 condizione iniziale, ma per le eq. di II grado serve imporre 2 condizioni: per esempio per calcolare il movimento di un oggetto nel tempo bisogna conoscere la sua velocità iniziale e la sua posizione iniziale. ($F = ma$ è di II grado perché $a$ è la derivata seconda della posizione)
\end{osservazione}

\subsubsection{Equazioni omogenee}
Per $f(t) \equiv 0$ ci conviene riprendere la funzione esponenziale sui complessi:
\begin{gather*}
     \text{Con }\lambda \in \mathbb{C}\ \ \ h(t) = e^{\lambda t} = e^{\lambda_1 t}(\cos(\lambda_2 t) + i\sin(\lambda_2 t)) \\
    \text{Si può dimostrare che: } h'(t) = \lambda e^{\lambda t} \text{ come nei reali.}
\end{gather*}

\begin{equation}
    y''(t) + a(t)y'(t) + b(t)y(t) = 0
\end{equation}
Preso
\begin{gather*}
    z(t): I\subseteq \mathbb{R}\longrightarrow \mathbb{C} \\
    z(t) = z_1(t) + iz_2(t)
\end{gather*}
Se $z(t)$ risolve (1.16)
\begin{gather*}
    z'(t) = z'_1(t) + iz'_2(t) \\
    z''(t) = z''_1(t) + iz''_2(t) \\
    z''(t) + a(t)z' + b(t)z = 0 \\
    z''_1(t) + iz''_2(t) + a(t)(z'(t) + iz_2'(t))+b(t)(z_1(t)+iz_2(t)) = 0 \\
    [z_1''(t)+a(t)z_1'(t)+b(t)z_1(t)] + i[z''_2(t)+a(t)z'_2(t)+b(t)z_2(t)] = 0 \\
\end{gather*}
Questo è vero s.se entrambe le quadre sono 0 cioè risolvono l'eq come funzioni reali. \\
Quindi trovata 1 soluzione complessa che risolve l'eq trovo \textit{2 soluzioni} nel campo reale ($z_1$ e $z_2$). \\
Però noi possiamo solo risolvere equazioni dove $a$ e $b$ sono costanti:
\begin{equation*}
    y''(t)+ay'(t)+by(t)=0
\end{equation*}
Applicando quello che abbiamo visto, dobbiamo cercare soluzioni del tipo
\begin{gather*}
    z(t) = e^{\lambda t}\ \ \ \lambda\in \mathbb{C} \\
    z'(t) = \lambda e^{\lambda t} \\
    z''(t) = \lambda^2e^{\lambda t}
\end{gather*}
nel campo complesso. Quindi:
\begin{gather*}
    z''+az'+bz = 0 \\
    \lambda^2e^{\lambda t}a \lambda^{\lambda t}+ be^{\lambda t} = 0 \\
    e^{\lambda t}[\lambda^2 + a \lambda + b] = 0
\end{gather*}

Per trovare le soluzioni dell'equazione differenziale di secondo ordine basta trovare le soluzioni dell'equazione di secondo grado nelle quadre:
\begin{equation}
    \lambda = \frac{-a \pm \sqrt{a^2 - 4ab}}{2} = \begin{cases}
        \Delta > 0 & \lambda_1, \lambda_2 \text{ soluzioni reali} \\
        \Delta = 0 & \lambda_1 = \lambda_2 \text{ soluzione reale} \\
        i\Delta < 0 & \lambda_1, \lambda_2 = \mu \pm iv \text{ soluzioni complesse}
    \end{cases}
\end{equation}
Queste saranno le soluzioni dell'equazione.

\begin{esempio}
    \begin{equation}
        y''+y' - 6y = 0
    \end{equation}
    Dobbiamo risolvere:
    \begin{gather*}
        \lambda^2 + \lambda - 6 = 0 \\
        \lambda = \frac{-1 \pm \sqrt{1+2 \cdot 4}}{2} = \begin{cases}
            \frac{-1 + 5}{2} = 2 \\
            \frac{-1 - 5}{2} = -3
        \end{cases}
    \end{gather*}
    Le soluzioni sono quindi
    \begin{align*}
        \lambda_2 &= -3 & \lambda_2 &= 2 \\
           &\downarrow    &    &\downarrow \\
           &e^{-3t} &    &e^{2t}
    \end{align*}
    Cioè
    \begin{equation}
        y(t)=c_1e^{-3t}+c_2e^{2t}.
    \end{equation}
\end{esempio}
