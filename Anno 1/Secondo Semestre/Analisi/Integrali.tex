\subsubsection{Lezione 5}
\section{Primitive e integrali indefiniti}
\begin{definizione}
    Data $f: I \subseteq \mathbb{R} \to \mathbb{R}$, una funzione $F$ è una \textit{primitiva} di $f$ in $I$ se:
    \begin{itemize}
        \item $F$ è derivabile in $I$
        \item $\frac{d}{dx}F(x)=f(x)$ in $I$
    \end{itemize}
\end{definizione}

\subsubsection{Problema}
Data $f$, determinare $F$. \\
Chiediamoci prima: quante primitive esistono? Con
\begin{gather}
    F \text{ primitiva in } [a,b] \to F'(x)=f \\
    G \text{ primitiva in } [a,b] \to G'(x)=f
\end{gather}
Allora
\begin{gather}
    \frac{d}{dx}(F(x)-G(x))= F'-G' = f-f= 0\ \ \forall x \in [a,b] \\
    F(x)-G(x)= \text{ costante per il teorema del valore di Lagrange} \\
    F(x)=G(x) + c\ \ c \in \mathbb{R}
\end{gather}
Quindi abbiamo che le primitive di $f$ sono:
\begin{equation}
    \{F(x)+c, c\in \mathbb{R}, F'=f\}
\end{equation}

\begin{teorema}[Teorema fondamentale del calcolo integrale]
    Sia $f$ continua in $[a,b]$. $F$ è una primitiva di
    \begin{gather*}
        f: \frac{dF}{dx} = f \\
        \iff \\
        F(x) = \int_{x_0}^{x} f(t) dt + c \\
        \text{dove } x_0 \in [a,b]
    \end{gather*}
\end{teorema}

\begin{dimostrazione}
    Sia $F: F' = f$
    \begin{equation}
        G(x) = \int_{x_0}^{x} f(t) dt
    \end{equation}
    $G$ è una primitiva perché $G' = f$. $F$ e $G$ differiscono per una costante:
    \begin{equation*}
        F(x) = G(x) + c
    \end{equation*}
    ???
\end{dimostrazione}

Come usiamo il teorema fondamentale del calcolo integrale (TFdCI) per gli integrali definiti: \\
\begin{gather}
    F(b) = \int_{x_0}^{b} f(t) dt + c \\
    F(a) = \int_{x_0}^{a} f(t) dt \\
    \mapsto F(b) - F(a) = \int_{x_0}^{b} f(t) dt = F(a) = \int_{x_0}^{a} f(t) dt =\\
    = \int_{x_0}^{b} f(t) dt + \int_{a}^{x_0} f(t) dt = \\
    = \int_{a}^{b} f(t) dt
\end{gather}
Questo era già valido per funzioni Lipschitziane o di classe 1, ma così è dimostrato per \textit{ogni funzione continua}.

\begin{definizione}[Integrale indefinito]
    L'insieme delle primitive di una funzione si chiama \textit{integrale indefinito} e si indica con
    \begin{equation}
        \int f(t) dt = \{F(t) + c, c\in \mathbb{R}\}
    \end{equation}
\end{definizione}

\begin{esempio}
    Polinomi:
    \begin{gather}
        \frac{d}{dx} \frac{x^{n+1}}{n+1}=x^n \Rightarrow \int x^n dx = \frac{x^{n+1}}{n+1}+c \\
        \frac{d}{dx} \sum_{k=0}^{n}a_{k}x^k=\sum_{k=0}^{n}a_k \frac{d}{dx}x^k = \sum_{k=0}^{n}a_k kx^{k-1}
    \end{gather}
\end{esempio}

\section{Metodi di integrazione}
\begin{proprieta}
    Dalla proprietà di derivazione di funzioni composte si ha che:
    \begin{equation}
        \int f(\alpha x) dx = \frac{\int f dx}{\alpha} + c
    \end{equation}
\end{proprieta}

Possiamo generalizzare:
\begin{equation}
    \frac{d}{dx}F(\Phi(x))= \frac{d}{dx}\Phi(x) \cdot \frac{d}{d\Phi}F(\Phi(x))
\end{equation}
Da cui:
\begin{equation}
    \int \frac{d}{dx}\Phi(x) \frac{d}{d\Phi}F(\Phi(x))dx = F(\Phi(x)) + c
\end{equation}

\begin{esempio}
    \begin{gather}
        \int 2x\cos(x^2)dx \\
        t = \Phi(x)=x^2 \mapsto dt = \Phi'(x)dx \\
        = \int \cos t dt = \sin t + c = \sin(x^2) +c
    \end{gather}
\end{esempio}

\subsection{Integrazione per sostituzione}
Se abbiamo un integrale del tipo:
\begin{equation}
    \int f(\Phi(x)) \Phi'(x) dx
\end{equation}
Possiamo risolverlo per sostituzione:

\begin{definizione}[Integrale per sostituzione]
    \leavevmode \\
    \begin{itemize}
        \item Sostituzione $ t = \Phi(x)$
        \item Sostituisco $dt = \Phi'(x)dx$
        \item L'integrale diventa:
    \end{itemize}
    \begin{equation}
        \int f(t) dt = F(t) + c = F(\Phi(x)) + c
    \end{equation}
\end{definizione}

Questo procedimento si può generalizzare anche per funzioni con esponenti reali:
\begin{equation}
    \int (\Phi(x))^{\alpha} \Phi'(x)dx = \int t^\alpha dt = \frac{t^{\alpha + 1}}{\alpha + 1} + c = \frac{(\Phi(x))^{\alpha + 1}}{\alpha + 1} + c
\end{equation}

\subsubsection{Lezione 6}
\subsection{Integrazione per parti}
Dal TFCI:
\begin{equation}
    \int_{a}^{b} \frac{d}{dx}f dx = f(b)-f(a) = [ f(x) ]^{x=b}_{x=a}
\end{equation}
Siano $f,g$ derivabili. Sappiamo che il prodotto di queste funzioni $fg$ è anch'esso derivabile:
\begin{equation}
    \frac{d}{dx}(fg) = \left(\frac{d}{dx}f\right)g + f \frac{d}{dx}g
\end{equation}
Integrando tra a e b:
\begin{gather}
    \int_{a}^{b} \frac{d}{dx}(fg) dx = \int_{a}^{b} \left[\left( \frac{d}{dx}f\right)g + f \frac{d}{dx}g\right] dx = \\
    = \int_{a}^{b} ( \frac{df}{dx})g dx + \int_{a}^{b} f \frac{d}{dx}g dx = \\
    = \int_{a}^{b} \frac{df(x)}{dx} dx = [f(x)g(x)]_a^b - \int_{a}^{b} f(x) \frac{dg(x)}{dx} dx
\end{gather}
Questa è la formula di \textit{integrazione per parti}.

\begin{definizione}[Formula di integrazione per parti]
    \begin{equation}
    \int_{a}^{b} g(x) \frac{df(x)}{dx} dx = [f(x)g(x)]_a^b - \int_{a}^{b} f(x) \frac{dg(x)}{dx} dx
    \end{equation}
\end{definizione}
\leavevmode \\

Con questa formula possiamo risolvere integrali del tipo:
\begin{itemize}
    \item $\int_{a}^{b} xe^{\alpha x} dx = \frac{1}{\alpha} \left(\left[xe^{\alpha x}- \frac{e^{\alpha x}}{\alpha}\right]_a^b\right)$
    \item $\int_{a}^{b} x\sin(x) dx = \left[-x\cos(x) + \sin(x)\right]_a^b$
    \item $\int_{a}^{b} x\cos(x) dx = \left[x\sin(x) + \cos(x)\right]_a^b$
    \item $\int_{a}^{b} \ln(x) dx = \int_{a}^{b} 1\cdot \ln(x) dx = \left[x(\ln(x) - 1\right]_a^b$
    \item $\int_{a}^{b} \arctan(x) dx = \int_{a}^{b} 1\cdot \arctan(x) dx = \left[x\arctan(x) - \ln\left(\sqrt{1+x^2}\right)\right]_a^b$
    \item ecc.
\end{itemize}
Questa formula vale anche per integrali indefiniti. \\
Vediamo alcuni integrali particolari:
\begin{gather}
    \int (\sin x)^k dx \\
    \int (\cos x)^k dx
\end{gather}
Si calcolano in modo diverso in base a $k$: se è pari si calcolano per parti; se è dispari si calcolano per sostituzione
\begin{enumerate}
    \item $k$ pari
        \begin{align*}
            &\int (cos x)^2 dx = \int \cos x \cdot \cos x dx = \\
            &= \int cos x \frac{d}{dx}\sin x dx = \\
            &= \cos x \cdot \sin x - \int \sin x \frac{d}{dx}\cos x dx = \\
            &= \cos x \cdot \sin x + \int (\sin x)^2 dx = \\
            &= \cos x \cdot \sin x + \int (1-\cos^2 x)dx = \\
            &= \cos x \cdot \sin x + x - \int (cos x)^2 dx = \\
            &=  2\int(cos x)^2 dx = \cos x\cdot \sin x + x + c \\
            &\int (cos x)^2 dx = \frac{\cos x\cdot \sin x + x + c}{2}
        \end{align*}
    \item $k$ dispari
        \begin{align*}
            &\int (cos x)^3 dx = \int(\cos x)^2 \cos x dx = \\
            &= \int(1-\sin^2 x) \cos x dx \xrightarrow{t = \sin x} \\
            &\rightarrow ... = \sin x - \frac{(\sin x)^3}{3} + c
        \end{align*}
\end{enumerate}

\subsubsection{Lezione 7}
\subsection{Integrazione di funzioni razionali}
Vediamo funzioni del tipo:
\begin{equation}
    \int \frac{P(x)}{Q(x)}dx \ \ \ \text{ con } P(x), Q(x) \text{ polinomi}
\end{equation}
Se $P(x)$ ha grado 0 si svolge per sostituzione. \\
\subsubsection{Numeratore di grado 1}
\begin{gather*}
    \int \frac{cx + d}{ax + b}dx \\
    \frac{cx + d}{ax + b} = A + \frac{B}{ax+b}\ \ \ \text{ con }A, B \text{ costanti da determinare} \\
    \int \frac{cx + d}{ax + b}dx = \int (A+ \frac{B}{ax + b})dx = Ax + \frac{B}{a}\ln|ax + b| + C \\
\end{gather*}
\begin{esempio}
    \begin{flalign*}
        &\int \frac{6x+8}{3x-5}dx \\
        &\frac{6x+8}{3x-5} = \frac{2\cdot [3x-5] + 10 + 8}{3x-5} = 2 + \frac{18}{3x-5} \\
        &\int \frac{6x+8}{3x-5}dx = \int (2+ \frac{18}{3x-5})dx =\\
        &= 2\int dx + 18\int \frac{dx}{3x-5}= 2x+18 \cdot \frac{1}{3}\ln|3x-5| + C
    \end{flalign*}
\end{esempio}

\subsubsection{Denominatore di grado 2}
\begin{equation*}
    \int \frac{dx}{Ax^2+Bx+C} = \frac{1}{A} \int \frac{dx}{x^2+ \frac{B}{A}x + \frac{C}{A}}
\end{equation*}
Si svolge in modo diverso in base alle radici del trinomio
\begin{enumerate}
    \item $\Delta = 0$
        \begin{equation}
            \int \frac{dx}{Ax^2+Bx+C} = \int \frac{dx}{(x+ \frac{b}{2})^2}=...= -\frac{1}{x+ \frac{b}{2}} + c
        \end{equation}
    \item $\Delta > 0$
        \begin{gather*}
            \frac{1}{x^2+bx+c}=...= \frac{A}{x-x_1}+ \frac{B}{x-x_2} \\
            \frac{x-x_1}{(x-x_1)(x-x_2)} = A + \frac{B}{x-x_2}(x-x_1) \\
            \xrightarrow{\lim_{x\to x_1}} \frac{1}{x_1-x_2} = A \\
            \text{boh} \\
            \int \frac{1}{x^2+bx+c} = A\int \frac{dx}{x-x_1}+ B\int \frac{dx}{x-x_2} = \\
            = \frac{1}{x_1-x_2}\ln|x-x_1| + \frac{1}{x_2-x_1}\ln|x-x_2|+ c = \\
            = \frac{1}{x_1-x_2}(\ln|x-x_1| - \ln|x-x_2|) + c = \\
            = \frac{1}{x_1-x_2}\ln|\frac{x-x_1}{x-x_2}|+ c
        \end{gather*}
    \item $\Delta < 0$
        \begin{gather*}
            x^2 + bx + c = (x-\alpha)^2 + \beta^2\ \ \ \text{ con alfa e beta reali da determinare} \\
            \int \frac{dx}{x^2 + bx + c} = \int \frac{dx}{(x-\alpha)^2 + \beta^2} \xrightarrow{t = x-\alpha} \int \frac{dt}{t^2+\beta^2} =\\
            = \frac{1}{\beta^2}\int \frac{dt}{\frac{t^2}{\beta^2}+1} \xrightarrow{u= \frac{t}{\beta}} boh \\
            = \frac{1}{\beta} \arctan \frac{x-\alpha}{\beta} + c
        \end{gather*}
\end{enumerate}

\subsubsection{Numeratore grado 1, denominatore grado 2}
\begin{equation}
    \int \frac{dx + e}{x^2+bx+c}dx
\end{equation}
\begin{enumerate}
    \item $\Delta > 0$
        \begin{gather*}
            x^2+bx+c = (x-x_1)(x-x_2) \\
            \frac{dx + e}{x^2+bx+c} = \frac{A}{x-x_1}+ \frac{B}{x-x_2} \\
            \int \frac{dx + e}{x^2+bx+c}dx = A\int \frac{dx}{x-x_1} + B\int \frac{dx}{x-x_2} = \\
            = A\ln|x-x_1| + B\ln|x-x_2| + c
        \end{gather*}
    \item $\Delta \leq 0$
        \begin{gather*}
            \frac{dx+f}{x^2+bx+c} = A\frac{2x+b}{x^2+bx+c}+ \frac{B}{x^2+bx+c} \\
            \text{Abbiamo che il secondo membro è un tipo che abbiamo già visto, il primo ha la derivata del denominatore al numeratore} \\
            \int \frac{dx + e}{x^2+bx+c}dx = A\int \frac{2x+b}{x^2+bx+c}dx + B\int\frac{dx}{x^2+bx+c} \\
            \text{sostituzione} \rightarrow = A\int \frac{dt}{t} + B\int ...\\
            = A\ln|x^2+bx+c|+ B\int \frac{dx}{x^2+bx+c}
        \end{gather*}
\end{enumerate}

\subsubsection{Lezione 8}
\section{Integrali Impropri}
Finora abbiamo visto integrali di funzioni limitate su un intervallo chiuso. Ma cosa succede se $f$ non è limitata, o consideriamo un intervallo di integrazione illimitato?

\begin{definizione}
    Sia $f: [a, x_0) \to \mathbb{R}$ continua in $[a, x_0)$ con
    \begin{equation*}
    \lim_{x\to x_0^-}f = \pm\infty.
    \end{equation*}
    È ben definito l'integrale
    \begin{equation*}
        \int_{a}^{x_0 - \varepsilon} f(x) dx
    \end{equation*}
    Se $\lim_{\varepsilon\to 0^+}\int_{a}^{x_0 - \varepsilon} f(x) dx = l\in \mathbb{R} \text{ finito}$ allora $f$ è \textit{integrabile in senso improprio} in $[a, x_0)$ e
    \begin{equation}
        \int_{a}^{x_0} f(x) dx = l.
    \end{equation}
\end{definizione}

\begin{corollario}
    Analogamente per $g: (x_0, b] \to \mathbb{R}$ continua in $(x_0, b]$ e con $\lim_{x\to x_0+}g(x) =\pm\infty$, se
    \begin{equation}
        \lim_{\varepsilon\to 0+}\int_{x_0+\varepsilon}^{b} g(x) dx= l
    \end{equation}
    Allora
    \begin{equation}
        \int_{x_0}^{b} g(x) dx = l.
    \end{equation}
\end{corollario}

\begin{esempio}
    Prendiamo
    \begin{equation}
        f(x) = \frac{1}{(x_0-x)^\alpha}, x\in[a,x_0)
    \end{equation}
    Si ha che $\int_{a}^{x_0} f(x) dx$ è finito s.se $\alpha < 1$. (vedi dimostrazione)
\end{esempio}

\begin{osservazione}
    Se $f \geq 0$ l'integrale improprio di $f$ o è finito o è infinito (come le serie).
\end{osservazione}

\begin{proprieta}[Criterio del Confronto]
    Siano $f,g$ continue in $[a, x_0)$. Se $0\leq f(x)\leq g(x)\ \ \forall x\in [a, x_0)$, allora
    \begin{equation}
        0 \leq \int_{a}^{x_0} f(x) dx \leq \int_{a}^{x_0} g(x) dx
    \end{equation}
    Quindi, in particolare:
    \begin{itemize}
        \item Se $g$ è integrabile in senso improprio allora anche $f$ lo è.
        \item Se l'integrale improprio di $f$ diverge, allora diverge anche quello di $g$.
    \end{itemize}
\end{proprieta}

\begin{esempio}
    \begin{equation}
        f(x) = \frac{\sqrt{7-x}}{\sqrt{3-x}}
    \end{equation}
    $f$ è integrabile in $(1,3)$? L'unico problema è in $x_0=3$: $f$ non è limitata. Abbiamo che
    \begin{equation}
        0 \leq \frac{\sqrt{7-x}}{\sqrt{3-x}} \leq \frac{\sqrt{6}}{\sqrt{3-x}}
    \end{equation}
    Vediamo che $\frac{\sqrt{6}}{\sqrt{3-x}}$ è integrabile in $[1,3)$, dato che è della forma $\frac{1}{(x_0-x)^\alpha}$ con $\alpha < 1\ \ (\alpha = -1/2)$, quindi per il confronto anche $f$ è integrabile.
\end{esempio}
\leavevmode \\

\begin{proprieta}[Criterio del Confronto Asintotico]
    Sia $f$ continua in $[a,x_0),\ f\geq 0$. Con
    \begin{equation}
        \lim_{x\to x_0^-}(x_0-x)^\alpha f(x) = l \not= 0, \text{ finito}
    \end{equation}
    Allora
    \begin{itemize}
        \item Se $\alpha < 1$ $f$ è integrabile in senso improprio.
        \item Se $\alpha \geq 1$ $f$ non è integrabile in senso improprio.
    \end{itemize}
\end{proprieta}

\subsection{Integrale su Intervalli Illimitati}
\begin{definizione}
    Sia $f$ continua in $[a, +\infty)$ (oppure $(-\infty, b]$). Se esiste finito $\lim_{c\to +\infty}\int_{a}^{c} f(x) dx = F(c)$ si dice che $f$ è integrabile in senso improprio in $[a, +\infty)$, e si definisce
    \begin{equation}
        \int_{a}^{+\infty} f(x) dx = \lim_{c\to +\infty}\int_{a}^{c} f(x) dx.
    \end{equation}
    Questo limite può essere finito o infinito.
\end{definizione}

\begin{osservazione}
    Se $f \geq 0$, allora $F(c)$ è monotona crescente.
\end{osservazione}

\begin{proprieta}[Criterio del Confronto]
    Siano $f,g$ continue in $[a, +\infty)$, $0\leq f(x)\leq g(x)$.
    Allora
    \begin{equation}
        0\leq \int_{a}^{+\infty} f(x) dx \leq \int_{a}^{+\infty} g(x) dx
    \end{equation}
    Quindi
    \begin{itemize}
        \item Se $\int_{a}^{+\infty} g(x) dx < +\infty$ allora $\int_{a}^{+\infty} f(x) dx < +\infty$.
        \item Se $\int_{a}^{+\infty} f(x) dx = +\infty$ allora $\int_{a}^{+\infty} g(x) dx = +\infty$.
    \end{itemize}
\end{proprieta}

\begin{proprieta}[Criterio del Confronto Asintotico]
    Sia $f\in [a, +\infty)$ continua non negativa. Se
    \begin{equation}
        \lim_{x\to +\infty}x^\alpha f(x) = l\not= 0 \text{ finito}
    \end{equation}
    Allora
    \begin{equation}
        f(x) \approx \frac{1}{x^\alpha}.
    \end{equation}
    Quindi:
    \begin{itemize}
        \item Se $\alpha > 1$ $f$ è integrabile in senso improprio: $\int_{a}^{+\infty} f(x) dx < \infty$.
        \item Se $\alpha \leq 1$ $f$ non è integrabile in senso improprio: $\int_{a}^{+\infty} f(x) dx = +\infty$.
    \end{itemize}
\end{proprieta}
