\section{Applicazioni lineari e matrici}
\subsubsection{Lezione 5}

\begin{proposizione}
Data un'applicazione lineare tra spazi vettoriali finitamente generati, scelta una base per il dominio e per il codominio, possiamo rappresentare l'applicazione lineare come matrice.
\end{proposizione}
\leavevmode \\
Sia $V\xrightarrow{f} W$ lineare. Prendiamo $\mathcal{B} = \{v_1,...,v_n\}$ base di $V$ e $\mathcal{C} = \{u_1,...,u_n\}$ base di $W$
\begin{definizione}
    $M_{\mathcal{C}}^{\mathcal{B}}$: la matrice associata a $f$ è definita così:
    \begin{equation}
        M_{\mathcal{C}}^{\mathcal{B}} = A = (a_{ij})\in M_{m,n}(\mathbb{K})
    \end{equation}

    Dove $A_j=X_{\mathcal{C}}(f(v_j))$, con $X_C(f(v_j)) :=$ colonna della $\mathcal{C}$-coordinata di $f(v_j)$
\end{definizione}

\begin{esempio}
    Prendiamo $\mathbb{K}=\mathbb{R},\ V=W=\mathbb{R}[x]_{\leq 2}$. Definiamo
    \begin{gather}
        \mathbb{R}[x]_{\leq 2}\xrightarrow{f} \mathbb{R}[x]_{\leq 2}
        p\mapsto p+p'
    \end{gather}
    ...
    \begin{equation}
        M_{\mathcal{C}}^{\mathcal{B}}(f)=\begin{bmatrix}
            0 & 1 & 0 \\
            0 & 1 & 2 \\
            0 & 0 & 1
        \end{bmatrix}
    \end{equation}
\end{esempio}

\begin{esempio}
    \begin{equation}
        V(\mathbb{E}^2)\xrightarrow{\rho_{\theta}} V(\mathbb{E}^2)
    \end{equation}
    Dove $\rho_{\theta}$ = "rotazione di angolo $\theta$" \\
    $\mathcal{B} = \mathcal{C} = $ base di $V(\mathbb{E}^2)$, detta base \textit{ortonaturale} = $\{\underline{i}, \underline{j}\}$
    \begin{equation}
        M_{\mathcal{B}}^{\mathcal{B}}(\rho_\theta) = \begin{bmatrix}
            \cos\theta & -\sin\theta \\
            \sin\theta & \cos\theta
        \end{bmatrix}
    \end{equation}
\end{esempio}

\begin{esempio}
    \begin{equation}
        \mathbb{K}^n \xrightarrow{L_A} \mathbb{K}^m
    \end{equation}

    Con $A\in M_{m,n}(\mathbb{K})$ \\
    \begin{gather}
        \mathcal{S}_n= \text{base standard di } \mathbb{K}^n \\
        \mathcal{S}_m= \text{base standard di } \mathbb{K}^m \\
        M_{\mathcal{S}_n}^{\mathcal{S}_m}(L_A)=A
    \end{gather}

    perché boh
\end{esempio}

\begin{proposizione}
    Se $v\in V$
    \begin{equation}
        X_{\mathcal{C}}(f(v))= M_{\mathcal{C}}^{\mathcal{B}}(f)\cdot X_{\mathcal{B}}(v)
    \end{equation}

    Praticamente se conosciamo la matrice associata ad $f$ conosciamo tutto di $f$
\end{proposizione}

\begin{dimostrazione}
    Siano $X_{\mathcal{B}}(v)=\left[\begin{smallmatrix}
        x \\ \vdots \\ x_n
    \end{smallmatrix}\right]$ \\
    Cioè $v=x_1v_1+...+x_nv_n$
    \begin{equation}
        f(v)=\sum_{j=1}^{n}x_jf(v_j)=x_1f(v_1)+x_2(fv_2)+...+x_nf(v_n)
    \end{equation}

    Poniamo
    \begin{equation}
        M_{\mathcal{C}}^{\mathcal{B}}(f) = (a_{ij})\in M_{n,m}
    \end{equation}
    \\
    Quindi
    \begin{equation}
        f(v_j)=a_{1j}w_1+a_{2j}w_2+...+a_{nj}w_n
    \end{equation}
    e perciò
    \begin{align}
         f(v) &= x_1 \sum_{i=1}^{m}a_{i1}w_i+x_2 \sum_{i=1}^{m}a_{i2}w_i+...+ x_n \sum_{i=1}^{m}a_{in}w_i = \\
        &= (\sum_{j=1}^{n}a_{1j}x_j)w_1 + (\sum_{j=1}^{n}a_{2j}x_j)w_2 + ... + (\sum_{j=1}^{n}a_{mj}x_j)w_m
    \end{align}

    Dove ogni sommatoria è uguale all'$n$-esima riga di $A\ (A^n)\cdot X$. \\
    In conclusione:
    \begin{equation}
        X_{\mathcal{C}}(f(v)) = \begin{bmatrix}
            A^1 \cdot X \\
            A^2 \cdot X \\
            \vdots \\
            A^n\cdot X
        \end{bmatrix} = A\cdot x = M_{\mathcal{C}}^{\mathcal{B}}(f)X_{\mathcal{B}}(v)
    \end{equation}
    C.V.D.
\end{dimostrazione}

Siano $V,W$ spazi vettoriali / $\mathbb{K}$
\begin{equation}
    \mathcal{L}(V,W) = \{V\xrightarrow{f} W\ |\ f \text{ lineare}\}
\end{equation}
Siano $f,g\in \mathcal{L}(V,W)$. Definiamo
\begin{notazione}
    \begin{gather}
        V\xrightarrow{f+g} W \\
        v\mapsto f(v)+g(v) \\
        V\xrightarrow{\lambda f} W \\
        \lambda f \mapsto \lambda f(v) \\
        V\xrightarrow{\underline{0}} W \\
        v\mapsto \underline{0}
    \end{gather}
\end{notazione}
$\mathcal{L}(V,W)$ con queste operazioni è uno spazio vettoriale / $\mathbb{K}$. L'elemento neutro è $\underline{0}$

\begin{proposizione}
    Ipotesi: $V,W$ finitamente generati, $\dim V = n$, $\dim W = m$ \\
    Tesi: se $\mathcal{B}$ base di $V$, $\mathcal{C}$ base di $W$, l'applicazione
    \begin{gather}
        \mathcal{L}(V,W) \xrightarrow{M_{\mathcal{C}}^{\mathcal{B}}} M_{m,n}(\mathbb{K}) \\
        f\mapsto M_{\mathcal{C}}^{\mathcal{B}}(f)
    \end{gather}
    È un isomorfismo.
\end{proposizione}

\begin{dimostrazione}
    È banale.
\end{dimostrazione}

\begin{proposizione}
    Siano
    \begin{equation}
        U\xrightarrow{g} V\xrightarrow{f} W
    \end{equation}
    Lineari, con $U,V,W$ finitamente generati, e siano
    \begin{itemize}
        \item $\mathcal{A} =$ base di $U$
        \item $\mathcal{V} =$ base di $V$
        \item $\mathcal{C} =$ base di $W$
    \end{itemize}

    Allora
    \begin{equation}
        M_{\mathcal{C}}^{\mathcal{A}}(f\circ g) = M_{\mathcal{C}}^{\mathcal{B}}(f) \circ M_{\mathcal{B}}^{\mathcal{A}}(g)
    \end{equation}
\end{proposizione}

\begin{dimostrazione}
    Sia $u\in U$.
    \begin{gather}
        X_{\mathcal{C}}(f\circ g(u)) = X_{\mathcal{C}}(f(g(u))) \\
        X_{\mathcal{C}}(f(v))=M_{\mathcal{C}}^{\mathcal{B}}(f)\circ X_{\mathcal{B}}(v) \\
        \downarrow \\
        X_{\mathcal{C}}(f(g(u))) = M_{\mathcal{C}}^{\mathcal{B}}(f) \cdot X_{\mathcal{B}}(g(u)) = M_{\mathcal{C}}^{\mathcal{B}}(f)\cdot M_{\mathcal{B}}^{\mathcal{A}}\cdot X_{\mathcal{A}}(u)
    \end{gather}
    Quindi
    \begin{equation}
        X_{\mathcal{C}}(f\circ g(u))=(M_{\mathcal{C}}^{\mathcal{B}}(f)\cdot M_{\mathcal{B}}^{\mathcal{A}}(g))\cdot X_{\mathcal{A}}(u)
    \end{equation}
    Ma sappiamo che
    \begin{gather}
        X_{\mathcal{C}}(f\circ g(u))=M_{\mathcal{C}}^{\mathcal{A}}(f\circ g)X_{\mathcal{A}}(u) \\
        \Downarrow \\
        M_{\mathcal{C}}^{\mathcal{B}}(f)\cdot M_{\mathcal{B}}^{\mathcal{A}}(g)=M_{\mathcal{C}}^{\mathcal{A}}(f\circ g)
    \end{gather}
\end{dimostrazione}

\subsubsection{Lezione 6}
\subsubsection{Problema}

Sia 
\begin{equation}
    V\xrightarrow{f} W
\end{equation}
Finitamente generata su $\mathbb{K}$. \newline
Trovare una base di
\begin{enumerate}
    \item $\im(f)\subset W$ \\
    \item $\ker f \subset V$
\end{enumerate}

Primo passo:
\begin{dimostrazione}
    Siano $\mathcal{B} = \{v_1,...,v_n\}$ e $\mathcal{C}= \{w_1,...,w_n\}$ basi di $V$ e $W$. Allora
    \begin{equation}
        M^{\mathcal{B}}_{\mathcal{C}}(f) = A\in M_{m,n}(\mathbb{K})
    \end{equation}
    E $\forall v\in V$
    \begin{equation}
        X_{\mathcal{C}}(f(v))=A\cdot X_B(v)
    \end{equation}

    Ricordiamo che
    \begin{equation}
        A_j=A\cdot \underline{e}_j = A\cdot \left[\begin{smallmatrix}
            0 \\ \vdots \\ 0 \\ 1 \\ 0 \\ \vdots \\ 0
        \end{smallmatrix}\right] = j
    \end{equation}
    Quindi
    \begin{equation}
        A_j= X_e(f(v_j))
    \end{equation}

    Conseguenza: per risolvere (1) estraggo una base di $\Span{A_1,...,A_n} \{A_{i1}, A_{i2},...,A_{ik}\}$ con $1\leq i_1 < i_2 < ... < i_j \leq i_k \Rightarrow$ Una base di $\im f$ è $\{f(v_{i1}),...,f(v_{ik})\}$. \newline
    Vogliamo quindi risolvere il problema:
    \begin{equation}
        \text{Data } A\in M_{m,n}(\mathbb{K})
    \end{equation}
    Trovare una base di $\Span{A_1,...,A_n}\subset \mathbb{K}^m$
\end{dimostrazione}

\begin{definizione}[Matrici a scala per colonna]
    Una matrice è a scala per colonna se, scorrendo le sue colonne, il primo valore non nullo della colonna ha indice inferiore del precedente.
\end{definizione}
\begin{esempio}[Matrice a scala per colonna]
    \begin{equation}
        \begin{bmatrix}
            3 & 0 & 0 \\
            0 & 1 & 0 \\
            0 & 1 & -1
        \end{bmatrix}
    \end{equation}
\end{esempio}

\begin{proprieta}
    Sia $A\in M_{m, n}(\mathbb{K})$ a scala per colonna. Una base di
    \begin{equation}
        \Span{A_1,...,A_n}
    \end{equation}
    è data dalla lista delle colonne non nulle.
\end{proprieta}

\subsubsection{Lezione 7}
\subsection{Diagonalizzazione}
Sia $V \xrightarrow{f} V$ un applicazione lineare. $f$ è diagonalizzabile se $\exists \mathcal{B} = \{v_1,...,v_n\}$ base di $V$ tale che:
\begin{equation}
    M_{\mathcal{B}}(f) = \begin{bmatrix}
        \lambda_1 & 0 & ... & 0 & 0 \\
        0  & \lambda_2 & ...
    \end{bmatrix}
\end{equation}

Dato $\lambda \in \mathbb{K}$:
\begin{gather}
    V_{\lambda}(f) = \ker (\lambda Id_V - f) \subset V \rightarrow \\
    V \xrightarrow{Id_V - f} V \\
    v \mapsto \lambda v-f(v)
\end{gather}
\begin{esempio}
    Supponiamo che $f$ sia diagonalizzabile e che valga (*). Se $\lambda \not \in \{\lambda_1,...,\lambda_n\}$, allora:
    \begin{equation}
        V_\lambda (f) = \{\underline{0}\}.
    \end{equation}
    Perché?
    \begin{equation}
        M_{\mathcal{B}}^{\mathcal{B}} (\lambda Id - f) = M^{\mathcal{B}}_{\mathcal{B}}( \lambda Id_V ) - M^{\mathcal{B}}_{\mathcal{B}}(f)
    \end{equation}
    Se $\lambda \in \{\lambda_1,...,\lambda_n\},\ V_{\lambda}(f)\not=\{\underline{0}\}$. 
\end{esempio}
Ricordiamo:
\begin{itemize}
    \item $\lambda\in \mathbb{K}$ è un \textit{autovalore di} $f$ se $V_{\lambda}(f)\not=\{\underline{0}\}.$
    \item $\underline{0}\not= v\in V$ è un \textit{autovettore di} $f$ se $\exists \lambda\in \mathbb{K}$ t.c. $v\in V_{\lambda}(f) \rightarrow f(v)=\lambda v$.
\end{itemize}

\begin{osservazione}
    $V \xrightarrow{f} V$ è diagonalizzabile s.se esiste una base di $V$ i cui elementi sono autovettori di $f$.
\end{osservazione}

\begin{notazione}
    $A\in M_{n,m}(\mathbb{K})$ è \underline{diagonalizzabile} se
    \begin{equation}
        \mathbb{K}^n \xrightarrow{L_A} \mathbb{K}^n
    \end{equation}
    è diagonalizzabile. Questo equivale a "$A$ è conuigata a una matrice diagonale". Inoltre:
    \begin{itemize}
        \item $\mathcal{S} =$ base standard di $\mathbb{K}$.
        \item $\mathcal{B} =$ base che diagonalizza $L_A$.
    \end{itemize}
    \begin{equation}
        \begin{bmatrix}
            \lambda_1 && 0 \\
                      & \ddots &\\
            0 && \lambda_n
        \end{bmatrix} = ...
    \end{equation}
\end{notazione}

Perché diagonalizzare? Supponiamo che $A\in M_{n,m}(\mathbb{K})$ sia diagonalizzabile:
\begin{gather}
    A=G^{-1}\cdot \Lambda \cdot G \\
    \text{con }\Lambda = \begin{bmatrix}
            \lambda_1 && 0 \\
            & \ddots &\\
            0 && \lambda_n
    \end{bmatrix}
\end{gather}
boh \\

Supponiamo che $A$ sia diagonalizzabile.
\begin{equation}
    A = G^{-1}\cdot \Lambda \cdot G
\end{equation}
Allora
\begin{gather}
    e^A = 1_n + G^{-1}\cdot \Lambda \cdot G + \frac{1}{2}G^{-1}\cdot \Lambda^2 \cdot G + \frac{1}{3}G^{-1}\cdot \Lambda^3 \cdot G + ... = \\
    = G^{-1}(1_n + \Lambda + \frac{1}{2} \Lambda ^2 + \frac{1}{3} \Lambda^3 + ...)\cdot G = boh = \\
    = \begin{bmatrix}
        e^{\lambda_1} && 0 \\
        & e^{\lambda_2} & \\
        0 && e^{\lambda_1}
    \end{bmatrix}
\end{gather}

\section{Determinanti}
\begin{definizione}
    Con $R$ anello commutativo con 1, un'applicazione lineare
    \begin{equation}
        M_{n,m}(R)\xrightarrow{\det_n} R
    \end{equation}
    Ricorsivamente:
    \begin{itemize}
        \item $\det_1((a))=a$
        \item Se $\det_{n - 1}$ è stato definito, poniamo
            \begin{equation}
                \det_n(A) := \sum_{j=1}^{n}(-1)^{n+j}\cdot a_{n,j}\cdot \det_{n-1}(A^n_j)
            \end{equation}
            Dove $a_{n,j}$ sono le entrate all'ultima riga di $A$ e $A^i_j$ è la matrice $(n-1) \times (n-1)$ ottenuta da $A$ eliminando riga $i$ e colonna $j$.
    \end{itemize}
\end{definizione}

\begin{esempio}
    \begin{equation}
        \det_2 \begin{bmatrix}
            a_{11} & a_{12} \\
            a_{21} & a_{22}
        \end{bmatrix} = -a_{21}\cdot a_{12} + a_{22} \cdot a_{11}=a_{11}\cdot a_{22} - a_{21}\cdot a_{12}
    \end{equation}
\end{esempio}

\begin{notazione}
    \begin{equation}
        \det A = |A|
    \end{equation}
\end{notazione}

\begin{proprieta}
    Sia $A\in M_{n,n}(R)$ con una colonna nulla. Allora $|A| = 0$.
    \begin{dimostrazione}
        Per induzione su $n$.
        \begin{equation}
            n=1 \ \ \ \det ((a)) = a = 0.
        \end{equation}
        Passo induttivo: \\
        \begin{gather}
            \text{Assumiamo vero per matrici }(n-1) \times (n-1)\\
            A = \begin{bmatrix}
                a_{11} & \cdots & 0 & \cdots \\
                       && \vdots && \\
                a_{1n}& \cdots & 0 & \cdots & a_{nn}
            \end{bmatrix}
        \end{gather}
        boh
    \end{dimostrazione}
\end{proprieta}

\begin{proprieta}
    Sia $A\in M_{n,n}(R)$ con $a_{ij}=0$ se $i<j$.
    \begin{equation}
        A = \begin{bmatrix}
            a_{11} \\
              & \ddots & \\
            * &   & a_{nn}
        \end{bmatrix}
    \end{equation}
    Allora $\det A=a_{11}\cdot a_{22}...\cdot a_{nn}$.
    \begin{dimostrazione}
        vedi
    \end{dimostrazione}
\end{proprieta}

\subsubsection{Calcolare determinanti}
Alcune proprietà:
\begin{enumerate}
    \item Se scambio due colonne diverse il determinante cambia segno.
    \item Se aggiungo a una colonna un multiplo di un'altra colonna il determinante non cambia.
    \item Se moltiplico una colonna per $\lambda$ il determinante viene moltiplicato per $\lambda$.
\end{enumerate}
\begin{esempio}
    \begin{equation}
        A =\begin{bmatrix}
            3 & 1 & 3 \\
            2 &-1 & 5 \\
            1 & 0 & 1
            \end{bmatrix} \mapsto \begin{bmatrix}
            1 & 3 & 3 \\
            -1& 2 & 5\\
            0 & 1 & 1
            \end{bmatrix} \mapsto \begin{bmatrix}
            1 & 0 & 0\\
            -1& 5 & 8 \\
            0 & 1 & 1
        \end{bmatrix}
    \end{equation}
    boh $\det A = 3$
\end{esempio}
