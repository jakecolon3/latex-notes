\chapter{Applicazioni lineari}\label{chap:applicazioni_lineari}
\begin{definizione*}[]

  Siano $V,W$ spazi vettoriali sullo stesso campo $\mathbb{K}$. Un'applicazione (una funzione) $f:V\to W$ è \textit{lineare} se:

  \begin{enumerate}
    \item $f(v_1+v_2)=f(v_1)+f(v_2) \text{ per } v_1,v_2\in V$
    \item $f(\lambda v)=\lambda f(v)$ per $\lambda\in \mathbb{K}$ e $v\in V$
  \end{enumerate}

\end{definizione*}

\begin{esempio*}[]

Sia
\begin{align}
  \mathbb{K}^n & \xrightarrow{f} \mathbb{K}\\
  (x_1,\dots,x_n) & \mapsto \sum_{i=1}^{n}a_ix_i \nonumber
\end{align}

$f$ è un'applicazione lineare.
\end{esempio*}

\begin{dimostrazione}[Punto 1.]
Dobbiamo dimostrare:
  \begin{equation}
    f(X+Y) \overset{?}{=} f(X)+f(Y) \\
  \end{equation}

\newpage
  Abbiamo che:
  \begin{align*}
    f(X+Y)=\sum_{i=1}^{n}a_i(x_i+y_i) & = a_1(x_1+y_1)+a_2(x_2+y_2)+... \\
    f(X) + f(Y) = \sum_{i=1}^{n}a_ix_i+\sum_{i=1}^{n}a_iy_i & = a_1x_1 + a_1y_1 + a_2x_2 + a_2y_2 +... \\
    \text{Raccogliendo gli $a_i$: }& = a_1(x_1+y_1) + a_2(x_2+y_2) + ...
  \end{align*}
  Vediamo quindi che vale 1.

\end{dimostrazione}

\begin{dimostrazione}[Punto 2.]

  Dobbiamo dimostrare:

  \begin{equation}
    f(\lambda X) \overset{?}{=} \lambda f(X)\\
  \end{equation}

  Abbiamo che:
  \begin{align*}
    f(\lambda X) =\sum_{i=1}^{n}a_i\lambda x_i = \lambda a_1x_1 + \lambda a_2x_2 + ... & = \lambda(a_1x_1 + a_2x_2 + ...)\\
    \lambda f(X) =\lambda \sum_{i=1}^{n}a_ix_i & = \lambda(a_1x_1 + a_2x_2 + ...)
  \end{align*}
  Vediamo quindi che vale 2.

\end{dimostrazione}

\section{Matrici}

\begin{definizione}[]
  Una matrice $m\times n$ con entrate in $\mathbb{K}$ è un'applicazione (funzione):

  \begin{align*}
    \{(i,j)\ |\ 1\leq i\leq m,\ 1\leq j\leq n\}&\to \mathbb{K}
    \label{eq:def_matrice} \\
    (i,j) &\to a_{i,j}
  \end{align*}

\end{definizione}

\begin{notazione}[Matrice $m\times n\ \ A$]

  \begin{equation}
    A=\begin{bmatrix}
      a_{1,1} & a_{1,2} &... & a _{1,n} \\
      \vdots  & \vdots & &\vdots \\
      \dots   &  a _{i,j} &... \\
      \vdots  & \vdots & &\vdots \\
      a _{m,1} & a _{m,2} & ...
    \end{bmatrix} \leftarrow \text{riga i}
    \label{eq:notaz_matrice}
  \end{equation}

  \begin{equation}
    M _{m,n}(\mathbb{K}) = \{A | A \text{ matrice } m\times n \text{ con entrate in }\mathbb{K}\} \nonumber
    \label{eq:matrice_A2}
  \end{equation}

\end{notazione}

\subsection{Operazioni con matrici}

Somma:
\begin{align*}
  M _{m,n}(\mathbb{K}\times M _{m,n}(\mathbb{K}) &\to M _{m,n}(\mathbb{K}) \\
  ((a_{i,j})+(b_{i,j}))&\mapsto (c_{i,j}) \\
  \text{dove } c_{i,j} := a&_{i,j}+ b _{i,j}.
\end{align*}

Moltiplicazione per uno scalare $\lambda \in \mathbb{K}$:
\begin{equation}
  \lambda \cdot A = (b _{i,j}) \text{ dove }b _{i,j}=\lambda \cdot a _{i,j}
  \label{eq:molt_mat}
\end{equation}

\begin{definizione}[Moltiplicazione tra matrici]
\begin{align}
  & A = M _{m,n}(\mathbb{K}) \\
  & B = M _{n,p}(\mathbb{K})
\end{align}

  La matrice $A\cdot B$ è la matrice $m\times p\ C$ (con entrate in $\mathbb{K}$) la cui entrata $c _{i,h}\ \ 1\leq i \leq m\ \ 1\leq h\leq p$ è data da

  \begin{equation}
    c _{i,h} := \sum_{j=1}^{n}a _{i,j}\cdot b _{j,h}
    \label{eq:mat_mult_entry}
  \end{equation}
\end{definizione}
\begin{esempio}[Moltiplicazione tra matrici]

    & B \cdot A \text{ non è definita.}
  \end{align}
\end{esempio}

\begin{notazione}[$A\in M _{m,n}(\mathbb{K})$]
  $ A^i $ per $1\leq i\leq m$ è la riga $i$-esima di $A$. \\
  $A_j$ per $\leq j\leq n$ è la colonna $j$-esima di $A$.
  \begin{equation}
    A = \begin{bmatrix}
      A^1 \\
      A^2 \\
      \vdots \\
      A^m
    \end{bmatrix} = \begin{bmatrix}
    A_1 & a_2 & ... & A_n
    \end{bmatrix}
    \label{eq:rige_colonne}
  \end{equation}
\end{notazione}
\begin{equation}
  A\cdot B = \begin{bmatrix}
    A^1 & ... & A^n
  \end{bmatrix} \cdot \begin{bmatrix}
    B_1 & ... & B_n
  \end{bmatrix} = \begin{bmatrix}
    A^1 B_1 & A^1 B_2 & ... \\
            & \vdots \\
    ... & A^i B_j & ... \\
  \end{bmatrix}
  \label{eq:mat_riga_per_colonna}
\end{equation}

\begin{notazione}[]
  La matrice unita $n\times n$ è
  \begin{equation}
    1 = 1_n = \begin{bmatrix}
      1 & 0 \\
      0 & 1
    \end{bmatrix}
    \label{eq:mat_unita}
  \end{equation}

  Una matrice $A\in M _{n,m}(\mathbb{K})$ è scalare se $A=\lambda 1_n\ \lambda \in \mathbb{K}$. \\
  Una matrice $A\in M _{n,m}(\mathbb{K})$ è diagonale se $A=\begin{bmatrix}
    \lambda & 0 \\
    0 & \lambda 
  \end{bmatrix}$
\end{notazione}
\begin{proprieta}[Proprieta algebriche]
  \begin{align}
    (\lambda 1_n)\cdot A &= \lambda A = A\cdot (\lambda 1_n) \label{prop_alg_1} \\
    (A\cdot B) \cdot C &= A\cdot (B\cdot C) \label{prop_alg_2} \\
    A\cdot (B+B') & = A\cdot B + A\cdot B' \label{prop_alg_3} \\
    (B+B') \cdot C & = B\cdot C+B' \cdot C \label{prop_alg_4}
  \end{align}
\end{proprieta}
